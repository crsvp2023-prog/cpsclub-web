// Firestore Security Rules for CPS Club
// This allows server-side Admin SDK operations while protecting user data

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Default: deny all
    match /{document=**} {
      allow read, write: if false;
    }
    
    // Users collection - Server only (via Admin SDK)
    // Admin SDK requests have auth.uid == null and are identified by x-goog-iam-authority header
    match /users/{userId} {
      // Allow server-side writes (Admin SDK)
      allow write: if request.auth == null;
      // Allow users to read their own profile
      allow read: if request.auth.uid == userId;
    }
    
    // Members collection - Server only
    match /members/{document=**} {
      allow write: if request.auth == null;
      allow read: if request.auth != null;
    }
    
    // Match availability collection - Server only
    match /match_availability/{document=**} {
      allow write: if request.auth == null;
      allow read: if request.auth != null;
    }
    
    // Newsletter subscribers collection - Server only
    match /newsletter_subscribers/{document=**} {
      allow write: if request.auth == null;
      allow read: if false; // Server only, no client reads
    }
    
    // Predictions collection - Allow public reads and server writes
    match /predictions/{document=**} {
      allow read: if true; // Public can read predictions
      allow write: if request.auth == null; // Server-side API writes only
    }
    
    // User votes collection - Allow server writes only
    match /userVotes/{document=**} {
      allow write: if request.auth == null; // Server-side API writes only
      allow read: if false; // No direct reads
    }
  }
}
